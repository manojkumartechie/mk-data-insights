            const projectGrid = document.getElementById('project-grid');
            const techFiltersContainer = document.getElementById('tech-filters');
            const domainFiltersContainer = document.getElementById('domain-filters');

            const allTechCategories = ['All', ...new Set(projects.map(p => p.techCategory))];
            const allFinancialDomains = ['All', ...new Set(projects.flatMap(p => p.financialDomains))];
            
            let activeTechFilter = 'All';
            let activeDomainFilter = 'All';

            function createFilterButtons(container, filters, type) {
                filters.forEach(filter => {
                    const button = document.createElement('button');
                    button.textContent = filter;
                    button.className = 'filter-btn px-3 py-1.5 text-sm font-medium rounded-full border';
                    button.dataset.filter = filter;

                    if (filter === 'All') {
                        button.classList.add('bg-blue-600', 'text-white', 'border-blue-600');
                    } else {
                        button.classList.add('bg-transparent', 'text-slate-300', 'border-slate-600', 'hover:bg-slate-700', 'hover:border-slate-500');
                    }

                    button.addEventListener('click', () => {
                        if (type === 'tech') {
                            activeTechFilter = filter;
                        } else {
                            activeDomainFilter = filter;
                        }
                        updateFilterButtons(type);
                        renderProjects();
                    });
                    container.appendChild(button);
                });
            }

            function updateFilterButtons(type) {
                const container = type === 'tech' ? techFiltersContainer : domainFiltersContainer;
                const activeFilter = type === 'tech' ? activeTechFilter : activeDomainFilter;
                
                container.querySelectorAll('.filter-btn').forEach(btn => {
                    btn.classList.remove('bg-blue-600', 'text-white', 'border-blue-600');
                    btn.classList.add('bg-transparent', 'text-slate-300', 'border-slate-600', 'hover:bg-slate-700', 'hover:border-slate-500');
                    if(btn.dataset.filter === activeFilter) {
                        btn.classList.add('bg-blue-600', 'text-white', 'border-blue-600');
                        btn.classList.remove('bg-transparent', 'text-slate-300', 'border-slate-600', 'hover:bg-slate-700', 'hover:border-slate-500');
                    }
                });
            }

            function renderProjects() {
                projectGrid.innerHTML = '';
                const filteredProjects = projects.filter(p => {
                    const techMatch = activeTechFilter === 'All' || p.techCategory === activeTechFilter;
                    const domainMatch = activeDomainFilter === 'All' || p.financialDomains.includes(activeDomainFilter);
                    return techMatch && domainMatch;
                });

                if(filteredProjects.length === 0){
                    projectGrid.innerHTML = `<div class="col-span-full text-center py-10"><p class="text-slate-500">No projects match the selected filters.</p></div>`;
                    return;
                }

                filteredProjects.forEach((p, index) => {
                    const card = document.createElement('div');
                    card.className = 'card glass-card rounded-xl p-6 cursor-pointer';
                    card.style.animationDelay = `${index * 50}ms`;
                    
                    card.innerHTML = `
                        <div class="flex justify-between items-start">
                             <h4 class="text-lg font-bold text-white mb-2">${p.title}</h4>
                             <div class="flex-shrink-0 ml-4 text-slate-400">
                                <svg xmlns="http://www.w3.org/2000/svg" class="h-6 w-6 transform transition-transform" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 9l-7 7-7-7" /></svg>
                             </div>
                        </div>
                        <p class="text-sm text-blue-400 mb-4">${p.techCategory}</p>
                        <div class="card-content">
                            <p class="text-slate-300 text-sm mb-4">${p.description}</p>
                            <div class="space-y-3 text-sm">
                                <div><strong class="font-semibold text-slate-200">Financial Domains:</strong> <span class="text-slate-300">${p.financialDomains.join(', ')}</span></div>
                                <div><strong class="font-semibold text-slate-200">Key Technologies:</strong> <span class="text-slate-300">${p.technologies.join(', ')}</span></div>
                                <div><strong class="font-semibold text-slate-200">User-Friendly Design:</strong> <span class="text-slate-300">${p.userFriendlyDesign}</span></div>
                                <div><strong class="font-semibold text-slate-200">Predictive Analytics:</strong> <span class="text-slate-300">${p.predictiveAnalytics}</span></div>
                            </div>
                        </div>
                    `;
                    card.addEventListener('click', () => {
                        card.classList.toggle('active');
                        card.querySelector('svg').classList.toggle('rotate-180');
                    });
                    projectGrid.appendChild(card);
                });
            }

            function renderSolutions() {
                const container = document.getElementById('solutions-grid');
                solutions.forEach(s => {
                    const item = document.createElement('div');
                    item.className = 'glass-card rounded-xl p-6';
                    item.innerHTML = `
                        <h4 class="text-xl font-bold text-white mb-3">${s.title}</h4>
                        <p class="text-slate-300 mb-4">${s.description}</p>
                        <div class="flex flex-wrap gap-2">
                            ${s.technologies.map(t => `<span class="bg-blue-500/20 text-blue-300 text-xs font-medium px-2.5 py-0.5 rounded-full">${t}</span>`).join('')}
                        </div>
                    `;
                    container.appendChild(item);
                });
            }

            function renderTrends() {
                const list = document.getElementById('trends-list');
                trends.forEach(t => {
                    const item = document.createElement('li');
                    item.textContent = t;
                    list.appendChild(item);
                });
            }

            function setupCharts() {
                Chart.defaults.color = '#94a3b8'; // slate-400
                Chart.defaults.borderColor = '#334155'; // slate-700

                const techData = projects.reduce((acc, p) => {
                    acc[p.techCategory] = (acc[p.techCategory] || 0) + 1;
                    return acc;
                }, {});

                const domainData = projects.reduce((acc, p) => {
                    p.financialDomains.forEach(domain => {
                        acc[domain] = (acc[domain] || 0) + 1;
                    });
                    return acc;
                }, {});

                const techChartCtx = document.getElementById('techChart').getContext('2d');
                new Chart(techChartCtx, {
                    type: 'bar',
                    data: {
                        labels: Object.keys(techData),
                        datasets: [{
                            label: '# of Projects',
                            data: Object.values(techData),
                            backgroundColor: 'rgba(59, 130, 246, 0.5)',
                            borderColor: 'rgba(59, 130, 246, 1)',
                            borderWidth: 1
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: { legend: { display: false }, tooltip: { backgroundColor: '#1e293b' } },
                        scales: { 
                            y: { beginAtZero: true, grid: { color: '#334155' } },
                            x: { grid: { color: '#334155' } }

            const navLinks = document.querySelectorAll('.nav-link');
            const sections = document.querySelectorAll('.app-section');
            const activeLinkClasses = ['text-white', 'border-blue-500'];
            const inactiveLinkClasses = ['text-slate-400', 'border-transparent'];
            
            function handleNavClick(e) {
                e.preventDefault();
                const targetId = e.target.dataset.section;
                
                sections.forEach(section => {
                    section.id === targetId ? section.classList.remove('hidden') : section.classList.add('hidden');
                });
                
                navLinks.forEach(link => {
                    link.classList.remove(...activeLinkClasses);
                    link.classList.add(...inactiveLinkClasses);
                });
                
                e.target.classList.add(...activeLinkClasses);
                e.target.classList.remove(...inactiveLinkClasses);

                window.scrollTo({top: 0, behavior: 'smooth'});
            }

            navLinks.forEach(link => link.addEventListener('click', handleNavClick));

            createFilterButtons(techFiltersContainer, allTechCategories, 'tech');
            createFilterButtons(domainFiltersContainer, allFinancialDomains, 'domain');
            renderProjects();
            renderSolutions();
            renderTrends();
            setupCharts();
            
            const initialLink = document.querySelector('.nav-link[data-section="explorer"]');
            initialLink.classList.add(...activeLinkClasses);
            initialLink.classList.remove(...inactiveLinkClasses);
        });
    </script>
